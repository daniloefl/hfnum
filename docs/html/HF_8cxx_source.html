411  // Calculating Y_k(orb1, orb2)[r]

426  // Z(r) = int_0^r Pk1*t Pk2*t (t/r)^k dt
427  // dZ/dr = d/dr[int Pk1*t Pk2*t (t/r)^k dt](r) - d/dr[int Pk1*t Pk2*t (t/r)^k dt](0)

432  // dZ/dr = Pk1*r Pk2*r - k/r Z
433  // d(rY)/dr = -k/r [int_0^r Pk1*t Pk2*t (t/r)^k dt] + (k+1)/r [int_r^inf Pk1*t Pk2*t (t/r)^k dt] for k >= 2
434  // --> for k < 2, there is a term from Pk1*t Pk2*t (r/t)^(k+1)|t = infinity
435  // d(rY)/dr = (k+1)/r (rY) - k/r Z - (k+1)/r Z
436  // d(rY)/dr = 1/r [ (k+1) (rY) - (2k + 1) Z ]
437  // Z (r=0) = 0
438  // lim Y when r -> infinity = Z

441  // d(rY)/dx = [ (k+1) (rY) - (2k + 1) Z ]
442  // d(exp(kx)*Z)/dx = k exp(kx) Z + exp(kx) dZ/dx

444  // d(exp(- (k+1)x) (rY))/dx = -(k+1) exp(-(k+1)x) (rY) + exp(-(k+1)x) d(rY)/dx

501 void HF::calculateVd(ldouble gamma) {
502  std::cout << "Calculating Vd." << std::endl;
503 
504  for (int k = 0; k < _o.size(); ++k) {
505  _vdsum[k] = Vradial(_g->N(), 0);
506  }
507  std::vector<bool> filled;
508  for (int k1 = 0; k1 < _o.size(); ++k1) {
509  if (_o[k1]->g() == 2*(2*_o[k1]->l()+1)) filled.push_back(true);
510  else filled.push_back(false);
511  }
512 
513  // calculate it first with filled orbitals, dividing by the number of orbitals
514  // this is exact if all 2(2*l+1) orbitals in this level are filled
515  for (int k1 = 0; k1 < _o.size(); ++k1) {
516  int l1 = _o[k1]->l();
517 
518  std::cout << "Calculating Vd term from k = " << k1 << std::endl;
519 
520  for (int k2 = 0; k2 < _o.size(); ++k2) {
521 
522  int l2 = _o[k2]->l();
523 
524  // from C. Fischer, "The Hartree-Fock method for atoms"
525  // Re-estimated in calculations/Angular coefficients Hartree-Fock numerical.ipynb
526  // Values agree, but taken in abs value ... how to average them in km?
527  // https://journals.aps.org/pr/pdf/10.1103/PhysRev.34.1293
528  ldouble C = 1.0;
529 
530  if (_averageCoefficients) {
531  // This is the central part
532  C = _o[k2]->g();
533  for (int ir1 = 0; ir1 < _g->N(); ++ir1) {
534  _vdsum[k1][ir1] += C*_Y[10000*0 + 100*k2 + 1*k2][ir1];
535  }
536  } else { // not average coefficients
537  if (filled[k2]) {
538  // This is the central part
539  C = _o[k2]->g();
540  for (int ir1 = 0; ir1 < _g->N(); ++ir1) {
541  _vdsum[k1][ir1] += C*_Y[10000*0 + 100*k2 + 1*k2][ir1];
542  }
543  } else { // not filled
544  C = _o[k2]->g();
545  for (int ir1 = 0; ir1 < _g->N(); ++ir1) {
546  _vdsum[k1][ir1] += C*_Y[10000*0 + 100*k2 + 1*k2][ir1];
547  }
548 
549  for (int k = 2; k <= 2*l2; k += 2) {
550  ldouble B = 0.0;
551  for (int ml1_idx = 0; ml1_idx < _o[k1]->term().size(); ++ml1_idx) {
552  int ml1 = ml1_idx/2 - l1;
553  if (_o[k1]->term()[ml1_idx] != '+' && _o[k1]->term()[ml1_idx] != '-') continue;
554  ldouble A = 0.0;
555  if (k == 2 && l2 == 1 && l1 == 1) {
556  for (int ml2_idx = 0; ml2_idx < _o[k2]->term().size(); ++ml2_idx) {
557  int ml2 = ml2_idx/2 - l2;
558  if (_o[k2]->term()[ml2_idx] != '+' && _o[k2]->term()[ml2_idx] != '-') continue;
559  if (ml1 == -1 && ml2 == -1) A += 1.0/25.0;
560  if (ml1 == -1 && ml2 == 0) A += -2.0/25.0;
561  if (ml1 == -1 && ml2 == 1) A += 1.0/25.0;
562  if (ml1 == 0 && ml2 == -1) A += -2.0/25.0;
563  if (ml1 == 0 && ml2 == 0) A += 4.0/25.0;
564  if (ml1 == 0 && ml2 == 1) A += -2.0/25.0;
565  if (ml1 == 1 && ml2 == -1) A += 1.0/25.0;
566  if (ml1 == 1 && ml2 == 0) A += -2.0/25.0;
567  if (ml1 == 1 && ml2 == 1) A += 1.0/25.0;
568  }
569  }
570  B += A;
571  }
572  // average over multiplicity of the k1, since the sum here is over the "other" orbitals
573  // we should not sum the contribution of "this" (k1) orbital more than once
574  B /= (ldouble) _o[k1]->g();
575 
576  if (B == 0) continue;
577  // This is the extra k parts
578  for (int ir1 = 0; ir1 < _g->N(); ++ir1) {
579  _vdsum[k1][ir1] += B * _Y[10000*k + 100*k2 + 1*k2][ir1];
580  }
581  }
582  }
583  } // if average coefficients
584 
585  }
586 
587  }
588 
589  for (int ko = 0; ko < _o.size(); ++ko) {
590  std::cout << "Adding Vd term for eq. " << ko << std::endl;
591  std::vector<ldouble> &currentVd = _vd[ko];
592  for (int k = 0; k < _g->N(); ++k) currentVd[k] = (1-gamma)*currentVd[k] + gamma*_vdsum[ko][k];
593  }
594 }
