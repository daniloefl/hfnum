523  int idx = _om.index(k);
524 
525  bool foundSolution = false;
526  if (iterE >= 2) { // if there are at least 3 points, fit a parabola
527  // f(E) = a E^2 + b E + c
528  // e = sum_k=0^3 (f(E_k) - F_k)^2
529  // grad e = 0
530  // de/da = 2 sum_k (f(E_k) - F_k) (E_k^2) = 2 a sum_k E_k^4 + 2 b sum_k E_k^3 + 2 c sum_k E_k^2 - sum_k F_k E_k^2
531  // de/db = 2 sum_k (f(E_k) - F_k) (E_k) = 2 a sum_k E_k^3 + 2 b sum_k E_k^2 + 2 c sum_k E_k - sum_k F_k E_k
532  // de/dc = 2 sum_k (f(E_k) - F_k) (1) = 2 a sum_k E_k^2 + 2 b sum_k E_k + 2 c sum_k 1 - sum_k F_k
533  VectorXld hE, hF;
534  hE.resize(3);
535  hF.resize(3);
536  hE(0) = _historyE[iterE-0](idx); hF(0) = _historyF[iterE-0](idx);
537  hE(1) = _historyE[iterE-1](idx); hF(1) = _historyF[iterE-1](idx);
538  hE(2) = _historyE[iterE-2](idx); hF(2) = _historyF[iterE-2](idx);
539  if ( ( (hF(0) > 0 && hF(1) > 0 && hF(2) > 0) ||
540  (hF(0) < 0 && hF(1) < 0 && hF(2) < 0) ) &&
541  (hF(0) - hF(1))/(hE(0) - hE(1)) >= 0.9*(hF(1) - hF(2))/(hE(1) - hE(2)) ) { // check if they are close to each other
542  MatrixXld A;
543  VectorXld F;
544  A.resize(3, 3);
545  F.resize(3);
546  A(0,0) = 2*(std::pow(hE(0), 4) + std::pow(hE(1), 4) + std::pow(hE(2), 4));
547  A(0,1) = 2*(std::pow(hE(0), 3) + std::pow(hE(1), 3) + std::pow(hE(2), 3));
548  A(0,2) = 2*(std::pow(hE(0), 2) + std::pow(hE(1), 2) + std::pow(hE(2), 2));
549  F(0) = hF(0)*std::pow(hE(0), 2) + hF(1)*std::pow(hE(1), 2) + hF(2)*std::pow(hE(2), 2);
550  A(1,0) = 2*(std::pow(hE(0), 3) + std::pow(hE(1), 3) + std::pow(hE(2), 3));
551  A(1,1) = 2*(std::pow(hE(0), 2) + std::pow(hE(1), 2) + std::pow(hE(2), 2));
552  A(1,2) = 2*(std::pow(hE(0), 1) + std::pow(hE(1), 1) + std::pow(hE(2), 1));
553  F(1) = hF(0)*std::pow(hE(0), 1) + hF(1)*std::pow(hE(1), 1) + hF(2)*std::pow(hE(2), 1);
554  A(2,0) = 2*(std::pow(hE(0), 2) + std::pow(hE(1), 2) + std::pow(hE(2), 1));
555  A(2,1) = 2*(std::pow(hE(0), 1) + std::pow(hE(1), 1) + std::pow(hE(2), 1));
556  A(2,2) = 2*(std::pow(hE(0), 0) + std::pow(hE(1), 0) + std::pow(hE(2), 0));
557  F(2) = hF(0)*std::pow(hE(0), 0) + hF(1)*std::pow(hE(1), 0) + hF(2)*std::pow(hE(2), 0);
558 
559  VectorXld p = A.inverse()*F;
560  ldouble a = p(0);
561  ldouble b = p(1);
562  ldouble c = p(2);
563  ldouble delta = b*b - 4*a*c;
564  if (delta > 0 && std::fabs(a) > 1e-6) {
565  ldouble E1 = (-b + std::sqrt(delta))/(2*a);
566  ldouble E2 = (-b - std::sqrt(delta))/(2*a);
567  if (E1 > 0 && E2 < 0) {
568  _dE[k] = E2 - _o[k]->E();
569  foundSolution = true;
570  } else if (E1 < 0 && E2 > 0) {
571  _dE[k] = E1 - _o[k]->E();
572  foundSolution = true;
573  } else if (E1 < 0 && E2 < 0) {
574  ldouble dE1 = E1 - _o[k]->E();
575  ldouble dE2 = E2 - _o[k]->E();
576  if (std::fabs(dE1) < std::fabs(dE2)) {
577  _dE[k] = dE1;
578  } else {
579  _dE[k] = dE2;
580  }
581  foundSolution = true;
582  }
583  }
584 
585  if (foundSolution) {
586  std::cout << "Orbital " << k << ", interpolated parabola using (E, F) = "
587  << "(" << _historyE[iterE](idx) << ", " << _historyF[iterE](idx) << "), "
588  << "(" << _historyE[iterE-1](idx) << ", " << _historyF[iterE-1](idx) << "), "
589  << "(" << _historyE[iterE-2](idx) << ", " << _historyF[iterE-2](idx) << ")"
590  << " --> found a,b,c = " << a << ", " << b << ", " << c << "; "
591  << "new E = " << _o[k]->E()+_dE[k] << ", dE = " << _dE[k] << std::endl;
592  }
593  } else { // if the (E, F) points do not have the same sign, then don't even try the parabola fit
594  std::cout << "INFO: Not trying the parabola fit, as the function to be optimised can vary too much for orbital " << k << std::endl;
595  }
596  }
597 
598  if (!foundSolution) {
599  dE[k] = E[k]*1e-1/_o[k]->n();
600  if (iterE >= 1) { // take a test E variation from the previous tested points
601  dE[k] = _historyE[iterE-1][k] - _historyE[iterE][k];
602  }
603 
604  std::vector<ldouble> EdE = E;
605  EdE[k] += dE[k];
606 
607  VectorXld Fd; // recalculate the function being minimized, because we are changing energy for orbital k independently from the others
608  if (_isSpinDependent) {
609  Fd = _iss.solve(EdE, _pot, _vsum_up, _vsum_dw, matchedSt);
610  } else {
611  Fd = _iss.solve(EdE, _pot, _vd, _vex, matchedSt);
612  }
613 
614  _dE[k] = 0;
615  if (std::fabs(Fn(idx) - Fd(idx)) > 1e-16) {
616  _dE[k] = Fn(idx)*(-dE[k])/(Fn(idx) - Fd(idx));
617  }
618  _dE[k] *= -gamma;
619 
620  if (std::fabs(_dE[k]) > 0.5) _dE[k] = 0.5*_dE[k]/std::fabs(_dE[k]);
621  std::cout << "Orbital " << k << " (with secant method), dE = " << _dE[k] << " (probe dE = " << dE[k] << ")" << std::endl;
622  }
623 
624  if (checkE) {
625  if (_nodes[k] < _o[k]->n() - _o[k]->l() - 1) {
626  std::cout << "Too few nodes in orbital " << k << ", skipping dE by large enough amount to go to the next node position." << std::endl;
627  _Emin[k] = _o[k]->E();
628  //_dE[k] = -_o[k]->E() + (_Emin[k] + _Emax[k])*0.5;
629  _dE[k] = _o[k]->E()*(_nodes[k] - _o[k]->n() + _o[k]->l() + 1)/(200.0*_o[k]->n());
630  std::cout << "Orbital " << k << ", new dE = " << _dE[k] << std::endl;
631  } else if (_nodes[k] > _o[k]->n() - _o[k]->l() - 1) {
632  std::cout << "Too many nodes in orbital " << k << ", skipping dE by large enough amount to go to the next node position." << std::endl;
633  _Emax[k] = _o[k]->E();
634  //_dE[k] = -_o[k]->E() + (_Emin[k] + _Emax[k])*0.5;
635  _dE[k] = _o[k]->E()*(_nodes[k] - _o[k]->n() + _o[k]->l() + 1)/(200.0*_o[k]->n());
636  std::cout << "Orbital " << k << ", new dE = " << _dE[k] << std::endl;
637  } else {
638  if (_dE[k] > 0) {
639  _Emin[k] = _o[k]->E();
640  } else if (_dE[k] < 0) {
641  _Emax[k] = _o[k]->E();
642  }
643  }
644  }
645  }
646 
647  VectorXld tmp;
648  tmp.resize(_o.size());
649  for (int k = 0; k < _o.size(); ++k) {
650  tmp(k) = _o[k]->E();
651  }
652  _historyE.push_back(tmp);
653  _historyF.push_back(Fn);
654 
655  ldouble F = Fn.norm();
656  return F;
657 }
