511  ldouble E0 = 0;
512  for (int k = 0; k < _o.size(); ++k) {
513  E0 += _o[k]->E();
514  }
515  ldouble J = 0;
516  ldouble vxcn = 0;
517  ldouble Exc = 0;
518  ldouble Ax = -3.0/4.0*std::pow(3.0/M_PI, 1.0/3.0);
519  for (int ir = 0; ir < _g->N(); ++ir) {
520  ldouble r = (*_g)(ir);
521  ldouble dr = 0;
522  if (ir < _g->N()-1)
523  dr = (*_g)(ir+1) - (*_g)(ir);
524  J += _u[ir]*(_n_up[ir] + _n_dw[ir])*std::pow(r, 2)*dr;
525  //vxcn += (_vex_lda_up[ir] + _vex_lda_dw[ir])*(_n_up[ir] + _n_dw[ir])*std::pow(r, 2)*dr;
526  vxcn += _vex_lda_up[ir]*_n_up[ir]*std::pow(r, 2)*dr;
527  vxcn += _vex_lda_dw[ir]*_n_dw[ir]*std::pow(r, 2)*dr;
528  //Exc += 0.5*Ax*std::pow(2*_n_up[ir] + 2*_n_dw[ir], 1.0/3.0)*std::pow(r, 2)*dr;
529  Exc += Ax*std::pow(_n_up[ir], 4.0/3.0)*std::pow(r, 2)*dr;
530  Exc += Ax*std::pow(_n_dw[ir], 4.0/3.0)*std::pow(r, 2)*dr;
531  }
532  E0 += -0.5*J - vxcn + Exc;
533  return E0;

613 void DFT::calculateV(ldouble gamma) {
614  std::cout << "Calculating u." << std::endl;
615  _u = std::vector<ldouble>(_g->N(), 0);
616 
617  ldouble Q = 0;
618  std::vector<ldouble> E(_g->N(), 0); // electric field
619  for (int ir1 = 0; ir1 < _g->N(); ++ir1) {
620  ldouble r1 = (*_g)(ir1);
621  ldouble dr = 0;
622  if (ir1 < _g->N()-1) dr = (*_g)(ir1+1) - (*_g)(ir1);
623  ldouble n_up = 0;
624  ldouble n_dw = 0;
625  n_up += _n_up[ir1];
626  n_dw += _n_dw[ir1];
627  Q += (n_up + n_dw)*std::pow(r1, 2)*dr;
628  E[ir1] = Q/std::pow(r1, 2);
629  }
630  _u[_g->N()-1] = Q/(*_g)(_g->N()-1);
631  for (int ir1 = _g->N()-2; ir1 >= 0; --ir1) {
632  ldouble dr = (*_g)(ir1+1) - (*_g)(ir1);
633  _u[ir1] = _u[ir1+1] + E[ir1]*dr;
634  }
635 
636  std::cout << "Calculating vex using LDA." << std::endl;
637  _vex_lda_up = std::vector<ldouble>(_g->N(), 0);
638  _vex_lda_dw = std::vector<ldouble>(_g->N(), 0);
639  // Ex = -3/4 (3/pi)^(1/3) int n^(4/3) dr = int exc n dr
640  // exc = -3/4 (3/pi)^(1/3) n^(1/3)
641  //
642  // vx = dE/dn
643  // with spin:
644  // Ex = 0.5*(Ex[2*n_up] + Ex[2*n_dw])
645  // vx = 0.5*(2*dE[n_up]/dn_up + 2*dE[n_dw]/dn_dw) = dE/dn_up + dE/dn_dw
646  // vx = exc + n dexc/dn
647 
648  // rs = (3/(4*pi*n))^(1/3)
649  // ex = -3/(4pi)*(3 pi^2 n)^(1/3) = -3/4 (3/pi)^(1/3) n^(1/3)
650  ldouble Ax = -3.0/4.0*std::pow(3.0/M_PI, 1.0/3.0);
651  for (int ir1 = 0; ir1 < _g->N(); ++ir1) {
652  //_vex_lda_up[ir1] += Ax*std::pow(_n_up[ir1] + _n_dw[ir1], 1.0/3.0);
653  //_vex_lda_dw[ir1] += Ax*std::pow(_n_up[ir1] + _n_dw[ir1], 1.0/3.0);
654 
655  //if (_n_up[ir1] + _n_dw[ir1] != 0) {
656  // _vex_lda_up[ir1] += (_n_up[ir1])*Ax*std::pow(_n_up[ir1] + _n_dw[ir1], -2.0/3.0);
657  // _vex_lda_dw[ir1] += (_n_dw[ir1])*Ax*std::pow(_n_up[ir1] + _n_dw[ir1], -2.0/3.0);
658  //}
659 
660  _vex_lda_up[ir1] += Ax*4.0/3.0*std::pow(_n_up[ir1], 1.0/3.0);
661  _vex_lda_dw[ir1] += Ax*4.0/3.0*std::pow(_n_dw[ir1], 1.0/3.0);
662 
663  //if (_n_up[ir1] + _n_dw[ir1] != 0) {
664  // _vex_lda_up[ir1] += (_n_up[ir1] + _n_dw[ir1])*Ax*std::pow(2*_n_up[ir1], -2.0/3.0);
665  // _vex_lda_dw[ir1] += (_n_up[ir1] + _n_dw[ir1])*Ax*std::pow(2*_n_dw[ir1], -2.0/3.0);
666  //}
667  }
668 
669  std::cout << "Calculating sum of SCF potentials." << std::endl;
670  for (int k = 0; k < _g->N(); ++k) _vsum_up[k] = (1-gamma)*_vsum_up[k] + gamma*(_u[k] + _vex_lda_up[k]);
671  for (int k = 0; k < _g->N(); ++k) _vsum_dw[k] = (1-gamma)*_vsum_dw[k] + gamma*(_u[k] + _vex_lda_dw[k]);
672 }
